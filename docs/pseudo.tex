  morecomment=[s]{/*}{*/},
  morestring=[b]",

  keywordstyle=[1]\color{cBlue1},
  keywordstyle=[2]\color{cRed},
  % Literates
  literate={:=}{{$\gets$}}1
  {<=}{{$\leq$}}1
  {>=}{{$\geq$}}1
  {<>}{{$\neq$}}1
  {!}{{$\neg$}}1
  {->}{{$\rightarrow$}}1
  {=>}{{$\Rightarrow$}}1
}
