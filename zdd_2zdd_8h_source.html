140  internal::cut_size_t add_false_cofactor(const internal::cut_type ct, const internal::cuts_t &ilevel_cuts) const
141  {
142  const safe_size_t cut_size = ilevel_cuts[ct];
143 
144  // Bit-mask (allowing implicit conversion to size_t with bit-operators) to
145  // get the cut-type WITHOUT the false arcs.
146  constexpr size_t bit_mask = internal::cut_type::INTERNAL_TRUE;
147  const internal::cut_type ct_excl_false = static_cast<internal::cut_type>(ct & bit_mask);
148 
149  // In product construction algorithms we need to take into account the
150  // (single) suppressed false arc, which may suddenly become visible (e.g.
151  // 'zdd_union'). Here, the DAG gets stuck inside of a copy of only one of
152  // the input ZDDs. To get there, one followed an arc of the input but
153  // afterwards one pairs with an invisible false arc that spans all levels.
154  //
155  // We do not need to account for this invisible false arc in the following
156  // two cases
157  //
158  // - If the requested cut does not include false arcs.
159  //
160  // - If the cut size is strictly larger than the corresponding cut_type
161  // excluding false. In this case, we already have a false arc to pair
162  // with.
163  const size_t add_suppressed = !includes_terminal(ct, false) && cut_size == ilevel_cuts[ct_excl_false];
164 
165  return to_size(cut_size + add_suppressed);
166  }
