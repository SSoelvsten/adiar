133  internal::cut_size_t add_false_cofactor(const internal::cut_type ct, const internal::cuts_t &ilevel_cuts) const
134  {
135  const safe_size_t cut_size = ilevel_cuts[ct];
136 
137  // Bit-mask (allowing implicit conversion to size_t with bit-operators) to
138  // get the cut-type WITHOUT the false arcs.
139  constexpr size_t bit_mask = internal::cut_type::INTERNAL_TRUE;
140  const internal::cut_type ct_excl_false = static_cast<internal::cut_type>(ct & bit_mask);
141 
142  // In product construction algorithms we need to take into account the
143  // (single) suppressed false arc, which may suddenly become visible (e.g.
144  // 'zdd_union'). Here, the DAG gets stuck inside of a copy of only one of
145  // the input ZDDs. To get there, one followed an arc of the input but
146  // afterwards one pairs with an invisible false arc that spans all levels.
147  //
148  // We do not need to account for this invisible false arc in the following
149  // two cases
150  //
151  // - If the requested cut does not include false arcs.
152  //
153  // - If the cut size is strictly larger than the corresponding cut_type
154  // excluding false. In this case, we already have a false arc to pair
155  // with.
156  const size_t add_suppressed = !includes_terminal(ct, false) && cut_size == ilevel_cuts[ct_excl_false];
157 
158  return to_size(cut_size + add_suppressed);
159  }
