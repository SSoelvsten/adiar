118  cut_size_t add_false_cofactor(const cut_type ct, const cuts_t &ilevel_cuts) const
119  {
120  const safe_size_t cut_size = ilevel_cuts[ct];
121 
122  // Bit-mask (allowing implicit conversion to size_t with bit-operators) to
123  // get the cut-type WITHOUT the false arcs.
124  constexpr size_t bit_mask = cut_type::INTERNAL_TRUE;
125  const cut_type ct_excl_false = static_cast<cut_type>(ct & bit_mask);
126 
127  // In product construction algorithms we need to take into account the
128  // (single) suppressed false arc, which may suddenly become visible (e.g.
129  // 'zdd_union'). Here, the DAG gets stuck inside of a copy of only one of
130  // the input ZDDs. To get there, one followed an arc of the input but
131  // afterwards one pairs with an invisible false arc that spans all levels.
132  //
133  // We do not need to account for this invisible false arc in the following
134  // two cases
135  //
136  // - If the requested cut does not include false arcs.
137  //
138  // - If the cut size is strictly larger than the corresponding cut_type
139  // excluding false. In this case, we already have a false arc to pair
140  // with.
141  const size_t add_suppressed = !includes_terminal(ct, false) && cut_size == ilevel_cuts[ct_excl_false];
142 
143  return to_size(cut_size + add_suppressed);
144  }
