126  internal::cut_size_t add_false_cofactor(const internal::cut_type ct, const internal::cuts_t &ilevel_cuts) const
127  {
128  const safe_size_t cut_size = ilevel_cuts[ct];
129 
130  // Bit-mask (allowing implicit conversion to size_t with bit-operators) to
131  // get the cut-type WITHOUT the false arcs.
132  constexpr size_t bit_mask = internal::cut_type::INTERNAL_TRUE;
133  const internal::cut_type ct_excl_false = static_cast<internal::cut_type>(ct & bit_mask);
134 
135  // In product construction algorithms we need to take into account the
136  // (single) suppressed false arc, which may suddenly become visible (e.g.
137  // 'zdd_union'). Here, the DAG gets stuck inside of a copy of only one of
138  // the input ZDDs. To get there, one followed an arc of the input but
139  // afterwards one pairs with an invisible false arc that spans all levels.
140  //
141  // We do not need to account for this invisible false arc in the following
142  // two cases
143  //
144  // - If the requested cut does not include false arcs.
145  //
146  // - If the cut size is strictly larger than the corresponding cut_type
147  // excluding false. In this case, we already have a false arc to pair
148  // with.
149  const size_t add_suppressed = !includes_terminal(ct, false) && cut_size == ilevel_cuts[ct_excl_false];
150 
151  return to_size(cut_size + add_suppressed);
152  }
