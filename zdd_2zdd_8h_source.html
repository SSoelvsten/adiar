168  internal::cut::size_type add_false_cofactor(const internal::cut ct, const internal::cuts_t &ilevel_cuts) const
169  {
170  const internal::safe_size_t cut_size = ilevel_cuts[ct];
171 
172  // Bit-mask (allowing implicit conversion to size_t with bit-operators) to
173  // get the cut-type WITHOUT the false arcs.
174  constexpr size_t bit_mask = internal::cut::Internal_True;
175  const internal::cut ct_excl_false = static_cast<internal::cut>(ct & bit_mask);
176 
177  // In product construction algorithms we need to take into account the
178  // (single) suppressed false arc, which may suddenly become visible (e.g.
179  // 'zdd_union'). Here, the DAG gets stuck inside of a copy of only one of
180  // the input ZDDs. To get there, one followed an arc of the input but
181  // afterwards one pairs with an invisible false arc that spans all levels.
182  //
183  // We do not need to account for this invisible false arc in the following
184  // two cases
185  //
186  // - If the requested cut does not include false arcs.
187  //
188  // - If the cut size is strictly larger than the corresponding cut
189  // excluding false. In this case, we already have a false arc to pair
190  // with.
191  const size_t add_suppressed = !ct.includes(false) && cut_size == ilevel_cuts[ct_excl_false];
192 
193  return internal::to_size(cut_size + add_suppressed);
194  }
