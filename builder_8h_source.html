201  builder_ptr<dd_policy> add_node(typename dd_policy::label_t label,
202  const builder_ptr<dd_policy> &low,
203  const builder_ptr<dd_policy> &high)
204  {
205  attach_if_needed();
206 
207  // Check validity of input
208  if(low.builder_ref != builder_ref || high.builder_ref != builder_ref) {
209  throw invalid_argument("Cannot use pointers from a different builder");
210  }
211  if(label > dd_policy::MAX_LABEL) {
212  throw invalid_argument("Nodes must have a valid label");
213  }
214  if(label > current_label) {
215  throw invalid_argument("Nodes must be added bottom-up");
216  }
217  if(low.uid.is_node() && low.uid.label() <= label) {
218  throw invalid_argument("Low child must point to a node with higher label");
219  }
220  if(high.uid.is_node() && high.uid.label() <= label) {
221  throw invalid_argument("High child must point to a node with higher label");
222  }
223 
224  // Update label and ID if necessary
225  if(label < current_label) {
226  current_label = label;
227  current_id = dd_policy::MAX_ID;
228  }
229 
230  // Create potential node
231  const node_t n(label, current_id, low.uid, high.uid);
232 
233  // Check whether this node is 'redundant'
234  const typename node_t::uid_t res_uid = dd_policy::reduction_rule(n);
235 
236  if (res_uid.is_terminal()) {
237  created_terminal = true;
238  terminal_val = res_uid.value();
239  }
240 
241  if(res_uid == low.uid) { return low; }
242  if(res_uid == high.uid) { return high; }
243 
244  // Push node to file
245  nw.push(n);
246  unref_nodes++;
247  current_id--;
248 
249  // Update count of unreferenced nodes
250  bool& low_unref = *low.unreferenced;
251  if(low_unref && !low.uid.is_terminal()) {
252  low_unref = false;
253  unref_nodes--;
254  }
255 
256  bool& high_unref = *high.unreferenced;
257  if(high_unref && !high.uid.is_terminal()) {
258  high_unref = false;
259  unref_nodes--;
260  }
261 
262  return make_ptr(res_uid);
263  }
