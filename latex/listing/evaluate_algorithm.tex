  Evaluate(x : bool[], G : OBDD):
    v := G.V[G.V.length - 1] // The root is outputted last by Reduce

    while True:
      if x[node.label]:
        case v.high of Sink{ value }        => return value
                     | Link{ index, label } => v := G.V[index]
      else:
        case v.low of Sink{ value }         => return value
                    | Link{ index, label }  => v := G.V[index]
