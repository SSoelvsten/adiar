524  inline ptr_uint64 essential(const ptr_uint64 &p)
525  {
526  // We can abuse the bit-layout to boil everything down to a bit mask, and a
527  // conditional move instruction. This should be optimisable to
528  // 'std::min<size_t>' into very few cpu instructions.
529  const uint64_t _raw = p._raw;
530 
531  constexpr uint64_t main_mask =
532  ~((1ull << ptr_uint64::FLAG_BITS) - 1u);
533 
534  constexpr uint64_t node_mask =
535  ~(((1ull << ptr_uint64::OUT_IDX_BITS) - 1u) << ptr_uint64::FLAG_BITS) & main_mask;
536 
537  return _raw > ptr_uint64::TERMINAL_BIT
538  ? (_raw & main_mask)
539  : (_raw & node_mask);
540  }
