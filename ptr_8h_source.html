542  inline ptr_uint64 essential(const ptr_uint64 &p)
543  {
544  // We can abuse the bit-layout to boil everything down to a bit mask, and a
545  // conditional move instruction. This should be optimisable to
546  // 'std::min<size_t>' into very few cpu instructions.
547  const uint64_t _raw = p._raw;
548 
549  constexpr uint64_t main_mask =
550  ~((1ull << ptr_uint64::flag_bits) - 1u);
551 
552  constexpr uint64_t node_mask =
553  ~(((1ull << ptr_uint64::out_idx_bits) - 1u) << ptr_uint64::flag_bits) & main_mask;
554 
555  return _raw > ptr_uint64::terminal_bit
556  ? (_raw & main_mask)
557  : (_raw & node_mask);
558  }
