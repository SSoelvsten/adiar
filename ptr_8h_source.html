628  inline ptr_uint64 essential(const ptr_uint64 &p)
629  {
630  // We can abuse the bit-layout to boil everything down to a bit mask, and a
631  // conditional move instruction. This should be optimisable to
632  // 'std::min<size_t>' into very few cpu instructions.
633  const uint64_t _raw = p._raw;
634 
635  constexpr uint64_t main_mask =
636  ~((1ull << ptr_uint64::flag_bits) - 1u);
637 
638  constexpr uint64_t node_mask =
639  ~(((1ull << ptr_uint64::out_idx_bits) - 1u) << ptr_uint64::flag_bits) & main_mask;
640 
641  return _raw > ptr_uint64::terminal_bit
642  ? (_raw & main_mask)
643  : (_raw & node_mask);
644  }
