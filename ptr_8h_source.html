525  inline ptr_uint64 essential(const ptr_uint64 &p)
526  {
527  // We can abuse the bit-layout to boil everything down to a bit mask, and a
528  // conditional move instruction. This should be optimisable to
529  // 'std::min<size_t>' into very few cpu instructions.
530  const uint64_t _raw = p._raw;
531 
532  constexpr uint64_t main_mask =
533  ~((1ull << ptr_uint64::FLAG_BITS) - 1u);
534 
535  constexpr uint64_t node_mask =
536  ~(((1ull << ptr_uint64::OUT_IDX_BITS) - 1u) << ptr_uint64::FLAG_BITS) & main_mask;
537 
538  return _raw > ptr_uint64::TERMINAL_BIT
539  ? (_raw & main_mask)
540  : (_raw & node_mask);
541  }
