370  void push(const node_t &n)
371  {
372  adiar_assert(attached(), "file_writer is not yet attached to any file");
373 
374  if (is_nil(_latest_node.uid)) { // First node pushed
375  _canonical = is_terminal(n) || id_of(n) == MAX_ID;
376  } else { // Check validity of input based on prior written node
377  adiar_debug(!is_terminal(_latest_node),
378  "Cannot push a node after having pushed a terminal");
379  adiar_debug(!is_terminal(n),
380  "Cannot push a terminal into non-empty file");
381 
382  // Check it is canonically sorted
383  if (_canonical) {
384  if (label_of(_latest_node) == label_of(n)) {
385  bool id_diff = id_of(n.uid) == id_of(_latest_node) - 1u;
386  bool children_ordered = n.high < _latest_node.high
387  || (n.high == _latest_node.high && n.low < _latest_node.low);
388 
389  _canonical = id_diff && children_ordered;
390  } else {
391  bool id_reset = id_of(n) == MAX_ID;
392  _canonical = id_reset;
393  }
394  }
395 
396  // Check if this is the first node of a new level
397  if (label_of(n) != label_of(_latest_node)) {
398  // Update level information with the level just finished
399  meta_file_writer::unsafe_push(create_level_info(label_of(_latest_node),
400  _level_size));
401  _level_size = 0u;
402 
403  // Update 1-level cut information
404  _max_1level_short_internal = std::max(_max_1level_short_internal,
405  _curr_1level_short_internal);
406 
407  _curr_1level_short_internal = 0u;
408  _long_internal_uid = create_node_uid(label_of(_latest_node), MAX_ID);
409  }
410  }
411 
412  // 1-level cut
413  const bool is_pushing_to_bottom = _long_internal_uid == NIL;
414  if (is_pushing_to_bottom && !is_terminal(n)) {
415  _terminals_at_bottom[value_of(n.low)]++;
416  _terminals_at_bottom[value_of(n.high)]++;
417  }
418 
419  if (is_node(n.low)) {
420  if (n.low > _long_internal_uid) { _number_of_long_internal_arcs++; }
421  else { _curr_1level_short_internal++; }
422  }
423 
424  if (is_node(n.high)) {
425  if (n.high > _long_internal_uid) { _number_of_long_internal_arcs++; }
426  else { _curr_1level_short_internal++; }
427  }
428 
429  // Update terminal counters
430  if (is_terminal(n.low)) { _file_ptr->number_of_terminals[value_of(n.low)]++; }
431  if (is_terminal(n.high)) { _file_ptr->number_of_terminals[value_of(n.high)]++; }
432  if (is_terminal(n.uid)) { _file_ptr->number_of_terminals[value_of(n.uid)]++; }
433 
434  // Write node to file
435  _latest_node = n;
436  _level_size++;
437 
438  meta_file_writer::unsafe_push(n, 0);
439  }

588  void fixup_ilevel_cuts()
589  {
590  const size_t number_of_nodes = meta_file_writer::size();
591  const size_t number_of_false = _file_ptr->number_of_terminals[false];
592  const size_t number_of_true = _file_ptr->number_of_terminals[true];
593 
594  // -----------------------------------------------------------------------
595  // Upper bound for any directed cut based on number of internal nodes.
596  const cut_size_t max_cut = number_of_nodes < MAX_CUT ? number_of_nodes + 1 : MAX_CUT;
597 
598  // -----------------------------------------------------------------------
599  // Upper bound on just 'all arcs'. This is better than 'max_cut' above, if
600  // there are 'number_of_nodes' or more arcs to terminals.
601  const bool noa_overflow_safe = number_of_nodes <= MAX_CUT / 2u;
602  const size_t number_of_arcs = 2u * number_of_nodes;
603 
604  const cuts_t all_arcs_cut = {
605  noa_overflow_safe ? number_of_arcs - number_of_false - number_of_true : MAX_CUT,
606  noa_overflow_safe ? number_of_arcs - number_of_true : MAX_CUT,
607  noa_overflow_safe ? number_of_arcs - number_of_false : MAX_CUT,
608  noa_overflow_safe ? number_of_arcs : MAX_CUT
609  };
610 
611  // -----------------------------------------------------------------------
612  // Maximum 1-level cuts
613  const bool is_terminal = number_of_false + number_of_true == 1;
614 
615  if (is_terminal) {
616  _file_ptr->max_1level_cut[cut_type::INTERNAL] = 0u;
617  _file_ptr->max_1level_cut[cut_type::INTERNAL_FALSE] = number_of_false;
618  _file_ptr->max_1level_cut[cut_type::INTERNAL_TRUE] = number_of_true;
619  _file_ptr->max_1level_cut[cut_type::ALL] = 1u;
620  } else {
621  for(size_t ct = 0u; ct < CUT_TYPES; ct++) {
622  // Use smallest sound upper bound. Since it is not a terminal, then there
623  // must be at least one in-going arc to the root.
624  _file_ptr->max_1level_cut[ct] = std::max(1lu, std::min({
625  _file_ptr->max_1level_cut[ct],
626  max_cut,
627  all_arcs_cut[ct]
628  }));
629  }
630  }
631 
632  // -----------------------------------------------------------------------
633  // Maximum 2-level cut
634  const size_t number_of_levels = meta_file_writer::levels();
635 
636  if (is_terminal || number_of_nodes == number_of_levels) {
637  for(size_t ct = 0u; ct < CUT_TYPES; ct++) {
638  _file_ptr->max_2level_cut[ct] = _file_ptr->max_1level_cut[ct];
639  }
640  } else { // General case
641  for(size_t ct = 0u; ct < CUT_TYPES; ct++) {
642  // Upper bound based on 1-level cut
643  const cut_size_t ub_from_1level_cut =
644  _file_ptr->max_1level_cut[ct] < MAX_CUT / 3u
645  ? ((_file_ptr->max_1level_cut[cut_type::INTERNAL] * 3u) / 2u
646  + (_file_ptr->max_1level_cut[ct] - _file_ptr->max_1level_cut[cut_type::INTERNAL]))
647  : MAX_CUT;
648 
649  // Use smallest sound upper bound.
650  _file_ptr->max_2level_cut[ct] = std::min({
651  _file_ptr->max_2level_cut[ct],
652  ub_from_1level_cut,
653  max_cut,
654  all_arcs_cut[ct]
655  });
656  }
657  }
658  }
