155  {

157  const shared_arcs_t &af = get<shared_arcs_t>();
158  return std::min(// 3/2 times the 1-level cut
159  (3 * af->max_1level_cut) / 2
160  + (includes_terminal(ct, false) ? af->number_of_terminals[false] : 0u)
161  + (includes_terminal(ct, true) ? af->number_of_terminals[true] : 0u),
162  // At most the number of nodes + 1
163  (af->size() / 2u) + 1);
164  }
165  if (has<shared_nodes_t>()) {
