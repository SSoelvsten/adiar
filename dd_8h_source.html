194  cut::size_type max_2level_cut(const cut ct) const
195  {
196  if (has<shared_arc_file_type>()) {
197  const shared_arc_file_type &af = get<shared_arc_file_type>();
198  return std::min(// 3/2 times the 1-level cut
199  (3 * af->max_1level_cut) / 2
200  + (ct.includes(false) ? af->number_of_terminals[false] : 0u)
201  + (ct.includes(true) ? af->number_of_terminals[true] : 0u),
202  // At most the number of nodes + 1
203  (af->size() / 2u) + 1);
204  }
205  if (has<shared_node_file_type>()) {
206  return get<shared_node_file_type>()->max_2level_cut[ct];
207  }
208  return 0u;
