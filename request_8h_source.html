356  template<size_t idx, typename Request>
357  struct request_lt
358  {
360  inline bool operator()(const Request &a, const Request &b)
361  {
362  const typename Request::label_type label_a = a.target.first().label();
363  const typename Request::label_type label_b = b.target.first().label();
364 
365  if constexpr (Request::cardinality == 2) {
366  constexpr size_t o_idx = 1u - idx;
367 
368  return label_a < label_b
369  || (label_a == label_b && a.target[idx] < b.target[idx])
370  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target[o_idx] < b.target[o_idx]);
371  }
372 
373  return label_a < label_b
374  || (label_a == label_b && a.target[idx] < b.target[idx])
375  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target < b.target);
376  }
377  };
