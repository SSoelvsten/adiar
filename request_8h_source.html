245  {
246  inline bool operator()(const request_t &a, const request_t &b)
247  {
248  const typename request_t::label_type label_a = a.target.first().label();
249  const typename request_t::label_type label_b = b.target.first().label();
250 
251  if constexpr (request_t::cardinality == 2) {
252  constexpr size_t o_idx = 1u - idx;
253 
254  return label_a < label_b
255  || (label_a == label_b && a.target[idx] < b.target[idx])
256  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target[o_idx] < b.target[o_idx]);
257  }
258 
259  return label_a < label_b
260  || (label_a == label_b && a.target[idx] < b.target[idx])
261  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target < b.target);
262  }
