243  struct request_lt
244  {
245  inline bool operator()(const request_t &a, const request_t &b)
246  {
247  const typename request_t::label_t label_a = a.target.first().label();
248  const typename request_t::label_t label_b = b.target.first().label();
249 
250  if constexpr (request_t::cardinality == 2) {
251  constexpr size_t o_idx = 1u - idx;
252 
253  return label_a < label_b
254  || (label_a == label_b && a.target[idx] < b.target[idx])
255  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target[o_idx] < b.target[o_idx]);
256  }
257 
258  return label_a < label_b
259  || (label_a == label_b && a.target[idx] < b.target[idx])
260  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target < b.target);
261  }
