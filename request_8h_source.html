250  {
251  inline bool operator()(const Request &a, const Request &b)
252  {
253  const typename Request::label_type label_a = a.target.first().label();
254  const typename Request::label_type label_b = b.target.first().label();
255 
256  if constexpr (Request::cardinality == 2) {
257  constexpr size_t o_idx = 1u - idx;
258 
259  return label_a < label_b
260  || (label_a == label_b && a.target[idx] < b.target[idx])
261  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target[o_idx] < b.target[o_idx]);
262  }
263 
264  return label_a < label_b
265  || (label_a == label_b && a.target[idx] < b.target[idx])
266  || (label_a == label_b && a.target[idx] == b.target[idx] && a.target < b.target);
267  }
